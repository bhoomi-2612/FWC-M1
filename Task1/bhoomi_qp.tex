\section*{Questions}

\begin{enumerate}
\setcounter{enumi}{13}

\item Evaluate:
\[
\int (3x + 5)\sqrt{5 + 4x - 2x^2}\, dx
\]

\item Solve the differential equation:
\[
x\frac{dy}{dx} + y - x + xy\cot x = 0,\; x \neq 0
\]

\item Solve the differential equation:
\[
(x^2 + 3xy + y^2)\, dx - x^2\, dy = 0
\]
given that $y=0$ when $x=1$.

\item Find the angle between the vectors $(\vec a+\vec b)$ and $(\vec a-\vec b)$ if
\[
\vec a = 2\hat{i} - \hat{j} + 3\hat{k}, \quad
\vec b = 3\hat{i} + \hat{j} - 2\hat{k}.
\]
Also find a vector perpendicular to both.

\item Show that the lines
\[
\frac{x-1}{3} = \frac{y-1}{-1} = \frac{z+1}{0}
\quad \text{and} \quad
\frac{x-4}{2} = \frac{y}{0} = \frac{z}{-3}
\]
intersect. Find their point of intersection.

\item A committee of 4 students is selected from 7 boys and 4 girls.
Find the probability that the committee contains exactly 2 girls,
given that at least one girl is selected.

\item Show that the relation $R$ defined by
\[
(a,b)R(c,d) \Rightarrow a+d = b+c
\]
on $A \times A$, where $A=\{1,2,3,\ldots,10\}$,
is an equivalence relation.
Hence write the equivalence class of $(3,4)$.

\item Solve:
\[
\mydet{
a+x & a-x & a \\
a-x & a+x & a-x \\
a & a-x & a+x
} = 0
\]

\item Show that the height of the cylinder of greatest volume
inscribed in a right circular cone of height $h$
and semi-vertical angle $\alpha$ is $\frac{h}{3}$.
Hence find the greatest volume:
\[
\frac{4}{27}\pi h^3 \tan^2\alpha
\]

\item Using integration, find the area of the triangle formed by
the negative $x$-axis and the tangent and normal to the circle
\[
x^2 + y^2 = 9
\]
at the point $(-1,2\sqrt{2})$.

\item Find the foot of the perpendicular and the perpendicular distance
from $P(4,3,2)$ to the plane
\[
x + 2y + 3z = 2
\]
Also find the image of $P$ in the plane.

\item A, B and C throw a pair of dice alternately until one gets a total of $9$.
Find the probability that A wins, if A starts first.

\item A company manufactures two types of cardigans.
Formulate the LPP and find the maximum profit graphically.

\end{enumerate}

\section*{Task 1}

\subsection*{Problem 1}

\solution
\[
x\frac{dy}{dx} + y - x + xy\cot x = 0
\]

\subsection*{Problem 2}

\solution
\[
(x^2 + 3xy + y^2)\,dx - x^2\,dy = 0, \quad y(1)=0
\]

\subsection*{Problem 3}

\solution
Angle between $(\vec a+\vec b)$ and $(\vec a-\vec b)$ is given by
\[
\cos\theta =
\frac{(\vec a+\vec b)\cdot(\vec a-\vec b)}
{\norm{\vec a+\vec b}\norm{\vec a-\vec b}}
\]
