\section*{\textbf{\Large CBSE Mathematics Examination, 2016}}

\begin{itemize}[label={},leftmargin=0pt]

\item \textbf{\large 2016/14.}
\item Evaluate:
\[
\int (3x + 5)\sqrt{5 + 4x - 2x^2}\, dx
\]

\item \textbf{\large 2016/15.}
\item Solve the differential equation:
\[
x\frac{dy}{dx} + y - x + xy\cot x = 0,\quad x \neq 0
\]

\item \textbf{\large 2016/16.}
\item Solve:
\[
(x^2 + 3xy + y^2)\, dx - x^2\, dy = 0
\]
given that $y=0$ when $x=1$.

\item \textbf{\large 2016/17.}
\item If 
\[
\vec a = 2\hat{i} - \hat{j} + 3\hat{k}, \quad
\vec b = 3\hat{i} + \hat{j} - 2\hat{k},
\]
find the angle between $(\vec a+\vec b)$ and $(\vec a-\vec b)$.  
Also find a vector perpendicular to both.

\item \textbf{\large 2016/18.}
\item Show that the lines
\[
\frac{x-1}{3} = \frac{y-1}{-1} = \frac{z+1}{0}
\quad \text{and} \quad
\frac{x-4}{2} = \frac{y}{0} = \frac{z}{-3}
\]
intersect. Find their point of intersection.

\item \textbf{\large 2016/19.}
\item A committee of 4 students is selected from 7 boys and 4 girls.  
Find the probability that the committee contains exactly 2 girls,
given that at least one girl is selected.

\item \textbf{\large 2016/20.}
\item Show that the relation $R$ defined by
\[
(a,b)R(c,d) \iff a+d = b+c
\]
on $A \times A$, where $A=\{1,2,\ldots,10\}$,
is an equivalence relation.  
Hence write the equivalence class of $(3,4)$.

\item \textbf{\large 2016/21.}
\item Solve:
\[
\begin{vmatrix}
a+x & a-x & a \\
a-x & a+x & a-x \\
a & a-x & a+x
\end{vmatrix}
=0
\]

\item \textbf{\large 2016/22.}
\item Show that the height of the cylinder of greatest volume inscribed in a right circular cone of height $h$ and semi-vertical angle $\alpha$ is $\frac{h}{3}$.  
Hence find the greatest volume.

\item \textbf{\large 2016/23.}
\item Using integration, find the area of the triangle formed by the negative $x$-axis and the tangent and normal to the circle
\[
x^2 + y^2 = 9
\]
at the point $(-1,2\sqrt{2})$.

\item \textbf{\large 2016/24.}
\item Find the foot of the perpendicular and the perpendicular distance from $P(4,3,2)$ to the plane
\[
x + 2y + 3z = 2.
\]
Also find the image of $P$ in the plane.

\item \textbf{\large 2016/25.}
\item A company manufactures two types of cardigans.  
Formulate the LPP and find the maximum profit graphically.

\end{itemize}
