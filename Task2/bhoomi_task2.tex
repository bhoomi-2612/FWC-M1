\section*{Task 2}

\begin{enumerate}

\item Solve the differential equation:
$$
\frac{dy}{dx} + y - x + xy \cot x = 0, \quad x \neq 0
$$

\item Solve the differential equation:
$$
(x^2 + 3xy + y^2)\,dx - x^2\,dy = 0
$$
Given that $y = 0$ when $x = 1$.

\item Find the angle between the vectors
$$
(\vec{a} + \vec{b}) \text{ and } (\vec{a} - \vec{b})
$$
if
$$
\vec{a} = 2\hat{i} - \hat{j} + 3\hat{k}, \quad
\vec{b} = 3\hat{i} + \hat{j} - 2\hat{k}.
$$
Also find a vector perpendicular to both.

\item Show that the lines
$$
\frac{x-1}{3} = \frac{y-1}{-1} = \frac{z+1}{0}
$$
and
$$
\frac{x-4}{2} = \frac{y}{0} = \frac{z}{-3}
$$
intersect. Find their point of intersection.

\item A committee of 4 students is selected from a group of 7 boys and 4 girls.
Find the probability that the committee contains exactly 2 girls,
given that at least one girl must be selected.

\end{enumerate}
